# 143. Reorder List

## Approach 

1. Find the middle of the list using fast and slow pointers.
2. Reverse the second half of the list.
3. Merge the two halves.

```java
class Solution {
    public void reorderList(ListNode head) {
        if (head == null || head.next == null) return;

        // 1. find middle
        ListNode middle = findMiddle(head);
        
        // 2. reverse right half;
        ListNode second = reverse(middle.next);
        middle.next = null; // cut the list into two halves;

        // 3. merge two halves
        ListNode first = head;
        while (first != null && second != null) {//也可以用dummy + tail 做
           ListNode temp1 =first.next;
           ListNode temp2 = second.next;

           first.next = second;
           first = temp1;

           second.next = temp1;
           second = temp2;
        }
    }

    private ListNode findMiddle(ListNode head) {
        ListNode slow = head, fast = head;

        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        return slow;
    }

    private ListNode reverse(ListNode head) {
        ListNode cur = head;
        ListNode pre = null, temp = null;

        while (cur != null) {
            temp = cur.next;
            cur.next = pre;
            pre = cur;
            cur = temp;
        }
        return pre;
    }
}
```
