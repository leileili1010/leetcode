# 695. Max Area of Island

## Approach - BFS

- time O(rows * cols);
- space O(rows * cols);

```java
class Solution {
    int rows;
    int cols;
    int[][] steps = {{0, 1}, {0, -1}, {-1, 0}, {1, 0}};
    
    public int maxAreaOfIsland(int[][] grid) {
        int max = 0;
        rows = grid.length;
        cols = grid[0].length;
        boolean[][] visited = new boolean[rows][cols];

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
               if (grid[i][j] == 1 && !visited[i][j]) {
                   int area = bfs(grid, i, j, visited); 
                   if (area > max) max = area;
               } 
            }
        }
        return max;
    }

    private int bfs(int[][] grid, int i, int j, boolean[][] visited) {
        Deque<int[]> que = new ArrayDeque<>();
        int area = 0;
        que.offer(new int[] {i, j});
        visited[i][j] = true;
        area++;

        while (!que.isEmpty()) {
            int[] node = que.poll();
            int row = node[0];
            int col = node[1];

            for (int[] step: steps){
                int newRow = row + step[0];
                int newCol = col + step[1];

                if (isValid(newRow, newCol) && grid[newRow][newCol] == 1 && !visited[newRow][newCol]) {
                    que.offer(new int[] {newRow, newCol});
                    visited[newRow][newCol] = true;
                    area++;
                }
            } 
        }
        return area;
    }

    private boolean isValid(int row, int col) {
        return row >= 0 && row < rows && col >= 0 && col < cols;
    }
}
```
