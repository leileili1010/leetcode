# 1029. Two City Scheduling
- time : Ologn
- space: o(1)

```java
// 推薦寫法
class Solution {
    public int twoCitySchedCost(int[][] costs) {
        // Sort by difference (costB - costA)
        Arrays.sort(costs, (a, b) -> (a[1] - a[0]) - (b[1] - b[0]));
        
        int n = costs.length / 2;
        int total = 0;
        
        // Send first n to city B, rest to city A
        for (int i = 0; i < costs.length; i++) {
            if (i < n) total += costs[i][1]; // cheaper to go B
            else total += costs[i][0];       // remaining go A
        }
        return total;
    }
}

// By Lei
class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int n = costs.length;
        int[][] diff = new int[n][2];
        int count  = 0;

        for (int i = 0; i < n; i++) {
            int[] cost = costs[i];
            diff[i][0] = cost[1] - cost[0];
            diff[i][1] = i;
        }

        Arrays.sort(diff, (a,b) -> Integer.compare(a[0], b[0]));

        for (int i = 0; i < n; i++) {
            if (i < n / 2) {
                count += costs[diff[i][1]][1];
            } else {
                 count += costs[diff[i][1]][0];
            }
        }

        return count;
    }
}
```
