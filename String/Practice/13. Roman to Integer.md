# 13. Roman to Integer

```java
class Solution {
    public int romanToInt(String s) {
        Map<Character, Integer> map = Map.of(
            'I', 1,
            'V', 5,
            'X', 10,
            'L', 50,
            'C', 100,
            'D', 500,
            'M', 1000
        );

        int res = 0;
        int pre = Integer.MAX_VALUE;
        for (char c: s.toCharArray()) {
            int cur = map.get(c);
            if (cur <= pre) {
                res += cur;
            } else {
                res += cur - 2 * pre;
            }
            pre = cur;
        }
        return res;  
    }
}
```
